152  @SuppressWarnings( "unchecked" )
153  public static Stream<RuntimeException> goaltrigger( final Stream<IAgent<?>> p_agents, final String p_action,
154  final String p_trigger, final String p_data, final boolean p_immediately )
155  {

163 
164  Stream.of(
165  "addgoal",

168  "deletebelief"

172  ( n ) -> CTrigger.from( ITrigger.EType.DELETEGOAL, n ),
173  ( n ) -> CTrigger.from( ITrigger.EType.ADDBELIEF, n ),
174  ( n ) -> CTrigger.from( ITrigger.EType.DELETEBELIEF, n )
175  )
176  ).map( i -> (ITrigger) i ).collect( Collectors.toSet() );
177 
178  return l_trigger.isEmpty()
179  ? Stream.of( new RuntimeException( CCommon.languagestring( CExecution.class, "actionunknown", l_action ) ) )
180  : p_agents.parallel()
181  .flatMap( i -> l_trigger.stream().map( j -> i.trigger( j, p_immediately ) ) )
182  .map( IFuzzyValue::value )
183  .filter( j -> !j )
184  .findAny()
185  .orElseGet( () -> true )
186  ? Stream.of()
187  : Stream.of( new RuntimeException( CCommon.languagestring( CExecution.class, "triggererror", l_action ) ) );
188  }

201  final Stream<Function<T, R>> p_executer )

220  private static <T> boolean actionconsumer( final String p_action, final Stream<T> p_values, final Stream<String> p_possibility,
221  final Stream<Consumer<Stream<T>>> p_executer )
222  {
223  return StreamUtils.zip(
224  p_possibility,
225  p_executer,

229  j.accept( p_values );

243  private static Stream<ITerm> parsestringterm( final String p_data )
244  {
245  return p_data.isEmpty()
246  ? Stream.of()
247  : Arrays.stream( p_data.split( ";|\\n" ) )
248  .map( String::trim )
249  .map( i -> parseterm(

252  CLiteral::parse,

254  ( j ) -> CRawTerm.from( Double.parseDouble( j ) )
255  ) );
256  }
